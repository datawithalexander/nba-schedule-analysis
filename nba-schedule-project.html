def plot_team_schedule(df, team, season):
    team_games = df.query("team == @team and season == @season").sort_values("gamedate")
    plt.figure(figsize=(14,2))
    plt.scatter(team_games["gamedate"], [1]*len(team_games), c="blue", marker="|", s=200)
    plt.title(f"{team} {season}-{season+1} Schedule")
    plt.yticks([])
    plt.xlabel("Date")
    plt.show()

import pandas as pd
import numpy as np
from sklearn.linear_model import LinearRegression

# --- Load data (expects the CSVs in the same folder as your notebook) ---
schedule = pd.read_csv("schedule.csv", parse_dates=["gamedate"])
game = pd.read_csv("team_game_data.csv", parse_dates=["gamedate"])

# --- Filter: Regular season 2019-20 through 2023-24 (season is the start year) ---
game_rs = game[(game["gametype"] == 2) & (game["season"] >= 2019) & (game["season"] <= 2023)].copy()

# --- Get opponent points by pairing the two team-rows of each game ---
# Each game has 2 rows with the same nbagameid: off_team A vs def_team B, and off_team B vs def_team A.
opp = game_rs[["nbagameid", "off_team", "points"]].rename(
    columns={"off_team": "opp_team", "points": "opp_points"}
)
game_rs = game_rs.merge(opp, on="nbagameid", how="left")
# Keep the opponent row that actually corresponds to this row’s defensive team
game_rs = game_rs[game_rs["opp_team"] == game_rs["def_team"]].copy()

# --- Compute rest days per team/date from schedule (reliable source for last game date) ---
sched_rs = schedule[(schedule["season"] >= 2019) & (schedule["season"] <= 2023)].copy()
sched_rs = sched_rs.sort_values(["team", "gamedate"])
sched_rs["prev_date"] = sched_rs.groupby("team")["gamedate"].shift(1)
sched_rs["rest_days"] = (sched_rs["gamedate"] - sched_rs["prev_date"]).dt.days
rest_lookup = sched_rs[["team", "gamedate", "rest_days"]]

# Join offensive team rest
game_rs = game_rs.merge(rest_lookup.rename(columns={"team": "off_team"}),
                        on=["off_team", "gamedate"], how="left")
game_rs = game_rs.rename(columns={"rest_days": "off_rest_days"})

# Join defensive team rest
game_rs = game_rs.merge(rest_lookup.rename(columns={"team": "def_team"}),
                        on=["def_team", "gamedate"], how="left")
game_rs = game_rs.rename(columns={"rest_days": "def_rest_days"})

# --- Feature flags ---
game_rs["off_b2b"] = (game_rs["off_rest_days"] == 1).astype(int)
game_rs["def_b2b"] = (game_rs["def_rest_days"] == 1).astype(int)
game_rs["off_home"] = game_rs["off_home"].astype(int)

# --- Target: off_team margin = points scored minus opponent points ---
game_rs["margin"] = game_rs["points"] - game_rs["opp_points"]

# --- Clean rows with all needed fields ---
model_df = game_rs.dropna(subset=["margin", "off_rest_days", "def_rest_days"]).copy()

# --- Fit a simple linear model for margin with schedule features only ---
X = model_df[["off_rest_days", "def_rest_days", "off_b2b", "def_b2b", "off_home"]].fillna(3)
y = model_df["margin"]
lin = LinearRegression().fit(X, y)

print("Model coefficients:", dict(zip(X.columns, lin.coef_)))
print("Intercept:", lin.intercept_)

# --- Compute schedule effect: real schedule vs neutral baseline (2 rest days, no B2B, neutral site) ---
neutral = model_df[["off_rest_days", "def_rest_days", "off_b2b", "def_b2b", "off_home"]].copy()
neutral["off_rest_days"] = 2
neutral["def_rest_days"] = 2
neutral["off_b2b"] = 0
neutral["def_b2b"] = 0
neutral["off_home"] = 0.5  # neutral site baseline

pred_real = lin.predict(X)
pred_neutral = lin.predict(neutral)

model_df["schedule_edge_margin"] = pred_real - pred_neutral

# --- Convert total margin to wins (~10 points of total margin ≈ 1 win) ---
team_wins_from_schedule = model_df.groupby("off_team")["schedule_edge_margin"].sum() / 10.0
team_wins_from_schedule = team_wins_from_schedule.sort_values()

# --- Results: most helped / most hurt by schedule over 2019-20 to 2023-24 ---
most_helped_team = team_wins_from_schedule.idxmax()
most_helped_value = float(team_wins_from_schedule.max())
most_hurt_team = team_wins_from_schedule.idxmin()
most_hurt_value = float(team_wins_from_schedule.min())

print("\nSchedule wins (total 2019-20 thru 2023-24):")
print(team_wins_from_schedule.round(2))

print("\nMost Helped by Schedule:", most_helped_team, f"({most_helped_value:.1f} wins)")
print("Most Hurt by Schedule:", most_hurt_team, f"({most_hurt_value:.1f} wins)")
